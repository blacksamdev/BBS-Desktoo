# BBS desktOO — dépendances
#
# Les planchers de version sont volontairement récents : un « >= » trop bas
# fait résoudre les scanners (Snyk, Dependabot) vers la plus vieille version
# acceptable, qui traîne d'anciennes CVE. On fixe donc des minima sûrs.

# --- Cœur (obligatoire) ---
PyQt6>=6.8.0
PyQt6-QScintilla>=2.14.1

# --- IA web embarquée (panneau haut : Claude, Gemini, Mistral) ---
# QtWebEngine embarque Chromium : garder une version récente est important.
PyQt6-WebEngine>=6.8.0

# --- Providers IA (optionnels selon usage) ---
anthropic>=0.40.0
openai>=1.40.0
httpx>=0.28.1

# --- Secrets dans le trousseau système (KWallet / GNOME Keyring) ---
keyring>=25.6.0
SecretStorage>=3.3.3

# --- Planchers de sécurité sur dépendances transitives ---
# desktOO n'importe pas ces paquets : ils sont tirés par httpx (anyio, idna)
# et par keyring -> SecretStorage (cryptography). Sans plancher explicite, la
# résolution des scanners retombe sur de vieilles versions vulnérables.
anyio>=4.4.0            # race condition (SNYK-PYTHON-ANYIO-7361842)
idna>=3.15              # ReDoS (SNYK-PYTHON-IDNA-16769942)
cryptography>=46.0.7    # authenticité / certificats / OOB write

# Chaîne keyring -> importlib_metadata -> zipp.
# keyring ne tire importlib_metadata que sur Python < 3.12 (au-delà,
# importlib.metadata est dans la stdlib). Or importlib_metadata ancien réclame
# « zipp>=0.5 », plancher si bas que les scanners résolvent zipp 3.15
# (boucle infinie, SNYK-PYTHON-ZIPP-7430899). On verrouille les deux niveaux :
# le parent (qui exige alors zipp>=3.20) et zipp lui-même, sans condition.
importlib_metadata>=8.5.0; python_version < "3.12"
zipp>=3.19.1
